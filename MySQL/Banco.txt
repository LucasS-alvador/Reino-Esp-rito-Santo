CREATE DATABASE IGREJA;
USE IGREJA;

-- Tabela de Missionários
CREATE TABLE MISSIONARIOS (
    ID INT AUTO_INCREMENT PRIMARY KEY,
    NOME VARCHAR(100),
    TELEFONE VARCHAR(15)
);

-- Tabela de Auxiliares
CREATE TABLE AUXILIARES (
    ID INT AUTO_INCREMENT PRIMARY KEY,
    NOME VARCHAR(100),
    FUNCAO VARCHAR(30),
    TELEFONE VARCHAR(15)
);

-- Tabela de Cultos
CREATE TABLE CULTOS (
    ID INT AUTO_INCREMENT PRIMARY KEY,
    DATA DATETIME,
    PASTORID INT,
    LINK VARCHAR(250),
    FOREIGN KEY (PASTORID) REFERENCES Missionarios(ID) -- Correção aqui
);

-- Tabela de Dízimos
CREATE TABLE DIZIMOS (
    ID INT AUTO_INCREMENT PRIMARY KEY,
    TOTAL DECIMAL,
    ADICIONADO DECIMAL,
    ANTIGOADD1 DECIMAL,
    ANTIGOADD2 DECIMAL,
    ANTIGOADD3 DECIMAL
);
